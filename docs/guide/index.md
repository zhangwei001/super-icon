## 安装
```jsx| pure
$ npm install @ali/super-icon --save

$ yarn add @ali/super-icon
```

## 使用
```jsx| pure
import { IconArrow } from "@ali/super-icon";

const App = () => (
     <IconArrow color={'#1E71FF'}  />
);
```
